# 文件创建和删除操作

# mkdir

# 基本使用
mkdir /file # 根目录创建 file 目录
mkdir ./file # 当前目录创建 file 目录
mkdir a b c # 建立多个目录
# 以上创建，目录不能存在，如果创建已存在目录会失败

# touch

修改文件或者目录的时间属性，包括存取时间和更改时间。若文件不存在，系统会建立一个新的文件。

touch [OPTION]... FILE...
-a    # 改变文件的读取时间
-c, --no-create # 如果文件不存在，不会建立新的文件
-d, --date=STRING # 设定时间和日期代替当前的时间
-m     # 改变文件的修改时间
-r, --reference=FILE # 使用FILE文件的时间记录
-t STAMP # 设定文件的时间记录，格式和date相同
# 创建一个文件
touch readme.md

# 文件的复制操作

# cp

默认情况下 cp命令只能复制文件，不能复制目录

# 用法
cp [OPTION]... [-T] SOURCE DEST
cp [OPTION]... SOURCE... DIRECTORY # 一般来说，这一种方法较为常用
cp [OPTION]... -t DIRECTORY SOURCE...
# 复制文件到 /tmp 目录，这是linux中的一个缓存目录
cp readme.md /tmp/
# 复制目录使用 选项 -r
cp -r filedir /tmp/
# 复制过程默认是没有过程的，通过-v来显示过程
cp -v readme.md /tmp/
# 复制过程中会改变文件的时间，通过 -p选项来保留 时间
cp -p filedir /tmp/
# 复制过程中会改变文件的权限信息，通过 -a选项来保留
cp -p -a filedir /tmp/

# mv

​ 负责文件的移动或者重命名

mv [OPTION]... -t DIRECTORY SOURCE...
# 将当前目录下的readme.md 移动到 当前目录下的 new_readme.md 移动的时候重命名

# 同样的可以对目录移动重命名

# 通配符

  • *匹配当前目录下所有的文件

    cp read* /tmp # 复制以read开头的所有文件
    1

  • ? 匹配单个字符

    cp read? /tmp # 复制以read开头，并且read后的字符只出现一次的所有文件
    1

# 打包和解压缩

# tar

再linux中，主要用来备份文件

# 常用选项
   -c, --create # 创建新文档
   -z, --gzip # gzip压缩
   -f # 指定操作类型为文件
   -x # 解压缩
   -C # change to directory DIR
   -t # 列出列出备份文件的内容。
   -v # 查看命令执行过程

用例

tar -cf archive.tar foo bar
# Create archive.tar from files foo and bar.

tar -tvf archive.tar
# List all files in archive.tar verbosely.

tar -xf archive.tar
# Extract all files from archive.tar.

tar -cf /tmp/test.tar ./new_dir # 打包 new_dir文件夹到tmp下的test.tar
tar -czf /tmp/test.tar.gz ./new_dir # 打包并压z
tar -xf /tmp/test.tar -C /home/ytt_test # 解压缩文件到指定目录
tar -xzf /tmp/test.tar.gz -C /home/ytt_test # 解压缩文件到指定目录
tar -tf /tmp/test.tar #查看压缩包文件内容
