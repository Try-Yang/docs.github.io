    # React.Children.map

    React.Children.map(this.props.children,function[(thisArg)])
    1

    这个函数的功能是在每一个children的子节点调用回调函数，然后返回一个新的children数组，如果children为null或者undefined，同样返回null或者undefined，而不会返回数组

    # 使用

    ​ 组合组件开发模式，一般用来解决父组件想要传递给字组件一些信息，但是又不想通过props来实现信息的传递，这样会使代码显得过于臃肿。（同时也不考虑context）

    ​ 在日常开发中，我们都碰到过这样的场景，假设有一个容器List，这个容器包含着多个对应的ListItem组件，我们点击不同的字组件会改变对应字组件的样式，或者携带对应字组件的id去触发ajax请求。

    ​ 一般情况下都会这样实现：

    let items = [
  {name:'小明',id:1},
  {name:'小红',id:2},
]
state = {
  currentIdx:0
}
<List>
    {items.map((item,idx)=>{
     	return <ListItem 
               {...item} 
               active={idx === this.state.currentIdx} 
               onClick={()=>{this.setState({currentIdx:idx})}}/>
    }))}    
</List>

    function ListItem(props){
  const {active,onClick} = props;
  const cls = active ? 'active' : ''
  const cb = useCallback(()=>{
    onClick(props) // 将传入的props传递会去，因为可能当前的回掉函数需要对应的已激活组件的信息
  },[])
  return (
  	<div className={cls} onClick={cb}>
    {props.children}
    </div>
  )
}

    上面我们在ListItem组件实现了active何onClick的处理，但是不通过props该如何将信息传递过去呢，这时候React.Children.map应运而生

    function List(props){
  const [activeIdx,setActiveIdx] = useState(0)
  const newChildren = React.Children.map(props.children,(child,idx)=>{
    if(child.type){ // 首先确保是一个元素
      return React.cloneElement(child,{
        active:this.state.active === activeIdx,
        onClick:(obj)=>{
          setActiveIdx(idx)
        }
      })
    }else{
      return child
  })
  return <>{newChildren}</>
}

    // packages/react/ReactChildren.js
function mapChildren(children,func,context){
  if(children === null){
    return children;
  }
  const result = [];
  let count = 0;
  mapIntoArray(children,result,"","",function(child){
    return func.call(context,child,count++)
  })
  return result;
}

    // 入参很多，暂时只关注children，和callback
function mapIntoArray(
	children,
  array,
  escapedPrefix,
  nameSoFar,
  callback)
{
    const type = typeof children;
	  if (type === 'undefined' || type === 'boolean') {
    children = null;
  	}
  	/*-----------*/
    let invokeCallback = false;
		if (children === null) {
    	invokeCallback = true;
    } else {
      switch (type) {
        case 'string':
        case 'number':
          invokeCallback = true;
          break;
        case 'object':
          switch ((children: any).$$typeof) {
            case REACT_ELEMENT_TYPE:
            case REACT_PORTAL_TYPE:
              invokeCallback = true;
          }
      }
    }
   /*------处理children使非数组的情况，也就是string，number，组件------*/
  if (invokeCallback) {
    const child = children;
    // 使用回调函数，处理children
    let mappedChild = callback(child);
    // 生成key，和主流程无关，不用在意
    const childKey =
      nameSoFar === '' ? SEPARATOR + getElementKey(child, 0) : nameSoFar;
    // 这里是个小彩蛋，如果我们的map回调最终返回了一个数组，
    if (Array.isArray(mappedChild)) {
      let escapedChildKey = '';
      if (childKey != null) {
        escapedChildKey = escapeUserProvidedKey(childKey) + '/';
      }
      // 那么需要扁平化这个数组，继续处理数组的每一个元素，并且添加到最终的result array中
      mapIntoArray(mappedChild, array, escapedChildKey, '', c => c);
    } else if (mappedChild != null) {
      // 如果不是一个数组，也就是最终的处理结果，判断是不是一个合法的element，如果是的化克隆一个新的对象出来添加到array中
      if (isValidElement(mappedChild)) {
        mappedChild = cloneAndReplaceKey(
          mappedChild,
          escapedPrefix +
            (mappedChild.key && (!child || child.key !== mappedChild.key)
                escapeUserProvidedKey('' + mappedChild.key) + '/'
              : '') +
            childKey,
        );
      }
      array.push(mappedChild);
    }
    return 1;
 	 }
  /*---------处理children的组件------------*/
  	let child;
  	let nextName;
  	let subtreeCount = 0;
  	const nextNamePrefix = nameSoFar === '' ? SEPARATOR : nameSoFar + SUBSEPARATOR;
		
  if (Array.isArray(children)) {
    // 处理数组，如果children是一个数组，那么就递归调用mapIntoArray，处理每一个元素，并且透传callback
    for (let i = 0; i < children.length; i++) {
      child = children[i];
      nextName = nextNamePrefix + getElementKey(child, i);
      subtreeCount += mapIntoArray(
        child,
        array,
        escapedPrefix,
        nextName,
        callback,
      );
    }
  } else {
    // children是一个迭代器，那么迭代执行这个迭代器，并且每执行一次都去处理一下他的value值
    const iteratorFn = getIteratorFn(children);
    if (typeof iteratorFn === 'function') {
      const iterableChildren: Iterable<React$Node> & {
        entries: any,
      } = (children: any);
      const iterator = iteratorFn.call(iterableChildren);
      let step;
      let ii = 0;
      while (!(step = iterator.next()).done) {
        child = step.value;
        nextName = nextNamePrefix + getElementKey(child, ii++);
        subtreeCount += mapIntoArray(
          child,
          array,
          escapedPrefix,
          nextName,
          callback,
        );
      }
    } else if (type === 'object') {
      const childrenString = '' + (children: any);
    }
  }
  return subtreeCount;
}
