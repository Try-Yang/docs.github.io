function render(
  element: React$Element<any>,
  container: Container,
  callback: ?Function,
){
    // --dev
    return legacyRenderSubtreeIntoContainer(
    null,
    element,
    container,
    false,
    callback,
  );
}

	parentComponent: ?React$Component<any, any>,
  children: ReactNodeList,
  container: Container,
  forceHydrate: boolean,
  callback: ?Function,
){
  let root = container._reactRootContainer
  let fiberRoot;
  if(!root){
    // 现在只分析这里，因为在第一次渲染的时候，container必然是没有root的
    root = container._reactRootContainer = legacyCreateRootFromDOMContainer(
      container,
      forceHydrate,
    );
    fiberRoot = root._internalRoot;
    // 暂停一下，先看看这个生成root做了什么
  }
}

# legacyCreateRootFromDOMContainer

function legacyCreateRootFromDOMContainer(container){
    const shouldHydrate =
    forceHydrate || shouldHydrateDueToLegacyHeuristic(container);
  // First clear any existing content.
  if (!shouldHydrate) {
    let warned = false;
    let rootSibling;
    // 主要是当前容器是否由子元素，有的话全部移除
    while ((rootSibling = container.lastChild)) {
      container.removeChild(rootSibling);
    }
  }
  return createLegacyRoot(
    container,
    undefined
  );
}

function createFiberRoot(
  containerInfo: any,
  tag: RootTag,
  hydrate: boolean,
  hydrationCallbacks: null | SuspenseHydrationCallbacks,
): FiberRoot {
    // 创建FiberRoot对象

    root.hydrationCallbacks = hydrationCallbacks;

# createHostRootFiber

function createHostRootFiber(tag: RootTag): Fiber {
  let mode;
  if (tag === ConcurrentRoot) {
    mode = ConcurrentMode | BlockingMode | StrictMode;
  } else if (tag === BlockingRoot) {
    mode = BlockingMode | StrictMode;
  } else {
    mode = NoMode;
  }
  return createFiber(HostRoot, null, null, mode);
}
const createFiber = function(
  tag: WorkTag,
  pendingProps: mixed,
  key: null | string,
  mode: TypeOfMode,
): Fiber {
  return new FiberNode(tag, pendingProps, key, mode);
};
function FiberNode(
  tag: WorkTag,
  pendingProps: mixed,
  key: null | string,
  mode: TypeOfMode,
) {
  // Instance
  this.tag = tag;

  this.sibling = null;
  this.index = 0;
  this.ref = null;
  this.pendingProps = pendingProps;
  this.memoizedProps = null;
  this.updateQueue = null;
  this.memoizedState = null;
  this.dependencies_new = null;
  this.mode = mode;
  // Effects
  this.effectTag = NoEffect;
  this.nextEffect = null;
  this.firstEffect = null;
  this.lastEffect = null;

  if (enableProfilerTimer) {

    this.treeBaseDuration = Number.NaN;
    this.actualDuration = 0;
    this.actualStartTime = -1;
    this.selfBaseDuration = 0;
    this.treeBaseDuration = 0;
  }
}

在创建createContainer的时候除了创建一个 FiberRoot，还创建了一个 RootFiber，同时两者互相引用，这两个对象的属性暂时不关注，针对

FiberRoot 目前需要了解到 的事 containerInfo和current 代表者容器信息和 RootFiber,对于RootFiber它的属性比FiberRoot要多很多了，核心去看一看child sibling这也是react16相比较16以下更改的地方，fiber 树构成了一个链表结构，父子节点通过return 值以及child互相关联，兄弟则通过 sibling关联

# 总结：

总结以下：在render的时候，首次渲染会先生成一个root对象，也就是FiberRoot，作用是什么先忽略，从 render函数开始到做最后 new FiberRootNode ,中间的其它函数都是为生成 FiberRoot铺垫，看源码之处，不比过于纠结每一个函数的作用，容易导致思维混乱 可以简化以下

let root = {
  _internalRoot: new FiberRootNode(container),
  render(){},
  unmount(){}
}
// 互相引用
root._internalRoot.currrent = new RootFiber(); // 生成RootFiber
root._internalRoot.currrent.stateNode = root._internalRoot;
